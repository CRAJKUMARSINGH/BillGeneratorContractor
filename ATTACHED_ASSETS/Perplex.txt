The BillGeneratorContractor repository focuses on a Streamlit Python app for generating contractor bills from PWD Schedule-G work order images and quantity text files, converting them to standard Excel input via OCR, but current OCR reliability is inconsistent (around 60% with plain Tesseract).[1][2]

## Key Enhancements Needed
Implement a foolproof OCR pipeline to achieve 92-96% accuracy and zero risk of silent wrong bills.[2]
- Add image preprocessing: grayscale conversion, Gaussian blur (5x5 kernel), adaptive thresholding.[2]
- Use grid-based table detection: morphological opening (40x1 kernel), contour filtering (width >600px, height >30px), row sorting by Y-coordinate.[2]
- Multi-mode row-wise OCR: Tesseract PSM 6/4/11 with OEM 3, auto-correct errors (O→0, l→1, S→5).[2]
- BSR code extraction via regex (pattern: \d+\.\d+\.\d+), quantity parsing from text files, strict validation (code matching, numeric rates, missing/extra items halt process).[2]

## Core Code Implementation
Add this to `modules/pwdscheduleparser.py` for the OCR engine:[2]

```python
import cv2
import pytesseract
import pandas as pd
import re

def preprocess(img_path):
    img = cv2.imread(img_path)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    blur = cv2.GaussianBlur(gray, (5, 5), 0)
    thresh = cv2.adaptiveThreshold(blur, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY, 11, 2)
    return thresh

def detect_rows(img):
    kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (40, 1))
    detect = cv2.morphologyEx(img, cv2.MORPH_OPEN, kernel)
    contours, _ = cv2.findContours(detect, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    rows = [cv2.boundingRect(c) for c in contours if cv2.boundingRect(c)[2] > 600 and cv2.boundingRect(c)[3] > 30]
    return sorted(rows, key=lambda r: r[1])

def ocr_rows(img, rows):
    lines = []
    for r in rows:
        x, y, w, h = r
        crop = img[y:y+h, x:x+w]
        txt = pytesseract.image_to_string(crop, config='--oem 3 --psm 6')
        txt = txt.replace('O', '0').replace('l', '1')
        lines.append(txt.strip())
    return lines

def parse_items(lines):
    data = []
    for line in lines:
        code_match = re.search(r'\d+\.\d+\.\d+', line)
        if code_match:
            parts = line.split()
            try:
                rate = float(parts[-3])
                qty = float(parts[-2])
                unit = parts[-4]
                desc = ' '.join(line.split(code_match.group(1), 1)[1]).strip()
                data.append([code_match.group(), desc, unit, rate, qty])
            except:
                continue
    return data

def process_workorder(img_path, excel_output):
    img = preprocess(img_path)
    rows = detect_rows(img)
    lines = ocr_rows(img, rows)
    data = parse_items(lines)
    if len(data) == 0:
        raise Exception("No Schedule-G items detected")
    df = pd.DataFrame(data, columns=['CODE', 'DESCRIPTION', 'UNIT', 'RATE', 'QTY'])
    df['AMOUNT'] = df['RATE'] * df['QTY']
    df.to_excel(excel_output, index=False)
```

Integrate into `app.py`: Call `process_workorder` on image upload, parse qty.txt, validate, then feed to existing bill engine.[2]

def validate_qty(qty_dict, codes):
    missing = set(qty_dict.keys()) - set(codes)
    if missing:
        raise Exception(f"Unknown item codes in qty file: {missing}")
