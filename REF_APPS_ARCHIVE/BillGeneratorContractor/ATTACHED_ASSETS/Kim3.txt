2.  Parallel Processing Architecture
from concurrent.futures import ProcessPoolExecutor, as_completed
import multiprocessing
def parallel_process_pages(pages, max_workers=4):
"""
Process 50 pages in parallel with controlled concurrency
"""
results = [None] * len(pages)  # Pre-allocate for ordering
with ProcessPoolExecutor(max_workers=max_workers) as executor:
    # Submit all tasks
    future_to_page = {
        executor.submit(process_single_page, page, i+1): i 
        for i, page in enumerate(pages)
    }
    
    # Process completions with progress tracking
    completed = 0
    for future in as_completed(future_to_page):
        page_idx = future_to_page[future]
        try:
            results[page_idx] = future.result()
            completed += 1
            print(f"Progress: {completed}/{len(pages)} pages ({completed/len(pages)*100:.1f}%)")
        except Exception as e:
            results[page_idx] = {'error': str(e), 'page': page_idx + 1}

return results

def validate_multi_page_extraction(all_pages_data, qty_dict):
"""
Comprehensive validation across all 49-50 pages
"""
all_codes = []
validation_report = {
'total_pages': 0,
'pages_with_data': 0,
'total_items': 0,
'missing_codes': [],
'duplicate_codes': [],
'rate_mismatches': [],
'qty_mismatches': []
}
Aggregate all codes
for page_data in all_pages_data:
validation_report['total_pages'] += 1
if page_data:
validation_report['pages_with_data'] += 1
for item in page_data:
all_codes.append(item['code'])
validation_report['total_items'] += 1
Check for duplicates (same item on multiple pages)
from collections import Counter
code_counts = Counter(all_codes)
duplicates = {code: count for code, count in code_counts.items() if count > 1}
if duplicates:
validation_report['duplicate_codes'] = duplicates
raise ValidationError(f"Duplicate item codes found: {duplicates}")
Validate against quantity file
qty_codes = set(qty_dict.keys())
extracted_codes = set(all_codes)
missing_in_qty = extracted_codes - qty_codes
if missing_in_qty:
validation_report['missing_codes'] = list(missing_in_qty)
raise ValidationError(f"Items in Work Order but not in Qty file: {missing_in_qty}")
extra_in_qty = qty_codes - extracted_codes
if extra_in_qty:
# Warning only - quantity file may have items from other work orders
validation_report['extra_qty_codes'] = list(extra_in_qty)
return validation_report

6.  Progress Tracking & User Feedback
import time
class ProgressTracker:
def init(self, total_pages):
self.total = total_pages
self.processed = 0
self.start_time = time.time()
self.page_times = []
def update(self, page_num, items_found):
    self.processed += 1
    elapsed = time.time() - self.start_time
    avg_time = elapsed / self.processed
    remaining = (self.total - self.processed) * avg_time
    
    print(f"[{self.processed}/{self.total}] Page {page_num}: "
          f"{items_found} items | "
          f"Elapsed: {elapsed//60:.0f}m {elapsed%60:.0f}s | "
          f"ETA: {remaining//60:.0f}m {remaining%60:.0f}s")

def summary(self):
    total_time = time.time() - self.start_time
    return {
        'total_pages': self.total,
        'processing_time': total_time,
        'pages_per_minute': self.total / (total_time / 60)
    }
